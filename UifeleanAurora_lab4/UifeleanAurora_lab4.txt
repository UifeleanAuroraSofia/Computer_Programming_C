/* UifeleanAuroraSofia, ETTI, Seria A, grupa 4
Program pentru calcularea unor sume din fractii subunitare (numere reale si intregi)
*/

#include <iostream>
using namespace std;
int main()
{
	float a, b, c;
	int d, e, f;
	cout << "Introduceti de la tastatura 3 numere reale: ";
	cin >> a >> b >> c;
	cout << 1 / a + 1 / b + 1 / c << endl;
	cout << "Introduceti de la tastatura 3 numere intregi: ";
	cin >> d >> e >> f;
	cout << 1 / d + 1 / e + 1 / f << endl;
	return 0;
}

/* UifeleanAuroraSofia, ETTI, Seria A, grupa 4
Program de efectuare a sumei ponderate dintre 3 numere intregi cu ponderi numere reale
a caror suma este 1
*/

#include<iostream>
using namespace std;

int main()
{
	int a, b, c;
	float p1, p2, p3;
	cin >> a >> b >> c;
	cin >> p1 >> p2 >> p3;
	cout << a * p1 + b * p2 + c * p3;
	return 0;
}
